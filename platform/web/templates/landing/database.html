query = db.query()
    .from_dimension('users')
    .where(active=True)
    .traverse('HAS_A', 'orders')
    .where(status='completed')
    .traverse('CONTAINS', 'products')
    .traverse('IS_IN', 'categories')
    .where(name='Electronics')
