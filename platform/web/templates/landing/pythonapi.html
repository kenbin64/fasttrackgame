from butterflyfx import PositionSubstrate, UniversalHardDrive

# Create a substrate
substrate = PositionSubstrate(
    expression_type='lambda',
    expression_code='lambda x, y: x * y'
)

# Store in Universal Hard Drive
hdd = UniversalHardDrive()
substrate_id = hdd.store(substrate, tags=['math', 'multiplication'])

# Retrieve and use
retrieved = hdd.retrieve(substrate_id)
result = retrieved.manifest(x=5, y=10)  # Returns 50

print(f"Result: {result}")  # Result: 50

Create Substrate

from butterflyfx import PositionSubstrate

# Create a mathematical substrate
substrate = PositionSubstrate(
    expression_type='lambda',
    expression_code='lambda x, y: x ** 2 + y ** 2'
)

# Manifest values
result = substrate.manifest(x=3, y=4)
print(result)  # 25 (3¬≤ + 4¬≤)

Store & Retrieve

from butterflyfx import UniversalHardDrive

hdd = UniversalHardDrive()

# Store substrate
substrate_id = hdd.store(
    substrate,
    tags=['math', 'geometry'],
    metadata={'author': 'user123'}
)

# Retrieve by ID
retrieved = hdd.retrieve(substrate_id)

# Query by tags
math_substrates = hdd.query(tags=['math'])

Relationships

from butterflyfx import RelationshipGraph, RelationType

graph = RelationshipGraph()

# Create relationships
graph.add_relationship(
    source_id=substrate1_id,
    target_id=substrate2_id,
    relation_type=RelationType.IS_PART_OF
)

# Query relationships
related = graph.get_related(
    substrate_id,
    relation_type=RelationType.HAS_A
)

Image Compression

from butterflyfx import ImageSubstrate

# Load image
image = ImageSubstrate.from_file('photo.jpg')

# Detect patterns
patterns = image.detect_patterns()

# Store as substrate (massive compression)
substrate_id = hdd.store(image)

# Retrieve and reconstruct
reconstructed = hdd.retrieve(substrate_id)
reconstructed.save('photo_reconstructed.jpg')
