Python ‚Äî Connect and Query

from helix import UniversalConnector

# Initialize and connect services
connector = UniversalConnector()
connector.connect("google_drive")
connector.connect("postgresql", config={"host": "localhost"})
connector.connect("claude_ai")

# Query across all connected services
results = connector.search("Q4 reports")
# ‚Üí Files from Drive, rows from PostgreSQL, all unified

# Cross-service operations
data = connector.get("postgresql://analytics/sales")
analysis = connector.analyze(data, service="claude_ai")
connector.save(analysis, "google_drive://Reports/q4_analysis.md")
